ERROR CHART¶

In [6]:
perceptron = Perceptron()

errors = list()
for i in range(1, 501):
    perceptron.n_iter = i
    perceptron.fit(X, y)
    errors.append(sum(perceptron.predict(X) != y))

plt.plot(range(1, len(errors) + 1), errors, color='r')
plt.xlabel('Iteration')
plt.ylabel('Error')

PETAL = [-((W[1] * sep) + W[0]) / W[2] for sep in SEPAL]
